### Installation

```
yarn add next-progressbar
```

### Usage

Edit your next config :

```js
// next.config.js
const withProgressBar = require('next-progressbar')

module.exports = withProgressBar({
  // rest of your next config
  // webpackbar options (optional)
  progressBar: {/* ... */}
})
```
